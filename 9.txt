I RN 1
J RN 2
K RN 3
A RN 4
AA RN 5
AAA RN 6
N RN 7
L RN 8
M RN 9
DUP RN 10
ENTRY
	LDR R0, =ARRAY
	MOV I, #0
	MOV N, #5
	SUBS M, N, #1
	MOV DUP, R0
PASS
	MOV J, #0
	SUBS K, N, I
	SUBS K, K, #1
SWAP
	LDR A, [R0]
	ADD L, R0, #4
	LDR AA, [L]
	CMP A, AA
	STRPL A, [L]
	STRPL AA, [R0]
	MOV R0, L
	ADD J, J, #1
	CMP J, K
	BNE SWAP
	ADD I, I ,#1
	CMP I, M
	MOV R0, DUP
	BNE PASS
GO	B GO
ARRAY DCD 9,1,4,3,5
	END
