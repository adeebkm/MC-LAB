	AREA NCR, CODE, READONLY
Q RN 3
REM RN 4
DV RN 1
DS RN 2
N RN 6
R RN 7
NUM RN 8
FACT RN 9
TEMP RN 10
ENTRY
	MOV N, #10 ;10C2
	MOV R, #2
	MOV NUM, N
	BL FACTORIAL
	MOV DV, FACT
	SUB NUM, N, R
	BL FACTORIAL
	MOV TEMP, FACT
	MOV NUM, R
	BL FACTORIAL
	MUL TEMP, FACT, TEMP
	MOV DS, TEMP
	BL DIV
STOP B STOP
FACTORIAL
	MOV FACT, #1
LOOP1
	MUL FACT, NUM, FACT
	SUBS NUM, NUM, #1
	BNE LOOP1
	MOV PC, LR
DIV
	MOV Q, #0
LOOP2
	SUBS DV, DV, DS
	ADDPL Q, Q, #1
	BPL LOOP2
	ADDMI REM, DV, DS
	MOV PC, LR
	END
